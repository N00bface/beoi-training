\subsection[Merge]{Merge sort}

\begin{frame}
\frametitle{Merge Sort}
\framesubtitle{Idee}
\begin{itemize}
\item Recursief de rij opdelen in 2 delen tot een stuk lengte 1 heeft
\item De twee gesorteerde delen samenvoegen tot 1 groot gesorteerd stuk
\end{itemize}
\end{frame}
