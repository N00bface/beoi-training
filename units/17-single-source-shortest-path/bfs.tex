\frame{
    \frametitle{BFS}
    \framesubtitle{Breath First Search}
    
    \begin{itemize}[<+->]
        \item Unweighted graphs
        \item Works with a queue (\emph{First in - First out})
        \item Take the next unexplored node from the frontier
        \item Goal reached: shortest path found
        \item Refer to \emph{Unit 14 - Graph traversals}
        \item $O(|V| + |E|)$
    \end{itemize}
}
